	agent any

			steps

				git 'https://github.com/elhadje87/IGP_2.git'
			}
		}

		{
			steps

				sh 'mvn compile'
			}
		}

		{
			steps
			{
				sh 'mvn test'
			}
		}

		stage('Code packaging')
		{
			steps
			{
				sh 'mvn package'
			}
		}
   }
}
