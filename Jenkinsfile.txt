pipeline
{
	agent any
	stages
	{
		stage('Code Checkout')
		{
			steps
			{
				git branch: 'main', url: 'https://github.com/elhadje87/IGP_2.git'
			}
		}
		
		stage('Code Compile')
		{
			steps
			{
				sh 'mvn compile'
			}
		}

		stage('Unit Test')
		{
			steps
			{
				sh 'mvn test'
			}
		}

		stage('Code packaging')
		{
			steps
			{
				sh 'mvn package'
			}
		}
		stage('Build Docker Image')
		{
			steps
			{
			    sh 'cp /var/lib/jenkins/workspace/abc-ci-cd/target/ABCtechnologies-1.0.war /var/lib/jenkins/workspace/abc-ci-cd/ABCtechnologies-1.0.war'
				
                sh 'docker build -t elhadje87/abc-ci-cd:$BUILD_NUMBER .'

			}
		}

		stage('Push Docker Image')
		{ 
			steps
			{   
			    withDockerRegistry([ credentialsId: "el-dockerhub", url: "" ])
			    {   
			       sh 'docker push elhadje87/abc-ci-cd:$BUILD_NUMBER'
				   
			    }
			}
		}

		stage('Deploy as container')
		{
			steps
			{
				sh 'docker run -itd -P elhadje87/abc-ci-cd:$BUILD_NUMBER'
			}
		}

   }
}
